function startProgram() {
  const n = document.getElementById('name-input').value.trim();
  if (!n) { alert('이름을 입력해주세요!'); return; }
  state.name = n;
  goToStep(1);
}

function goToStep(s) {
  state.step = s;
  const screens = ['welcome','story','read','quiz','play','result'];
  document.querySelectorAll('.screen').forEach(sc => sc.classList.remove('active'));
  document.getElementById('screen-' + screens[s]).classList.add('active');
  window.scrollTo(0, 0);
  if(s===1) initStory();
  if(s===2) initRead();
  if(s===3) initQuiz();
  if(s===4) initPlay();
  if(s===5) initResult();
}

    if(tn) tn.textContent = state.name;
    const total = state.quizScore + state.matchScore;
    if(cn) cn.textContent = '⭐ ' + total + '점';
  });
}

async function initQuiz() {
  updateTopBars();
  state.quizAnswers = {};
  document.getElementById('quiz-loading').style.display = 'block';
  document.getElementById('quiz-content').style.display = 'none';
  
  let questions = [];
  try {
    const res = await callClaude(`
당신은 한국어 교육 AI입니다. 다음 동화를 읽고 초등학교 1-3학년 다문화 아동을 위한 이해도 확인 문제 5개를 만들어주세요.

다음 JSON 형식으로만 응답해주세요 (다른 텍스트 없이):
[
  {"q":"질문","opts":["보기1","보기2","보기3","보기4"],"ans":정답인덱스(0-3)},
  ...
]
`);
    const clean = res.replace(/\`\`\`json|\`\`\`/g,'').trim();
    questions = JSON.parse(clean);
    if(!Array.isArray(questions)||questions.length<5) throw new Error('invalid');

  }

function renderQuiz(qs) {
  document.getElementById('quiz-loading').style.display = 'none';
  document.getElementById('quiz-content').style.display = 'block';
  const container = document.getElementById('quiz-questions');
  container.innerHTML = '';
  qs.forEach((q, i) => {
    const div = document.createElement('div');
    div.style.marginBottom = '24px';
    div.innerHTML = `<div class="section-title"><span class="q-num">${i+1}</span>${q.q}</div>
    <div class="quiz-options" id="opts-${i}">
      ${q.opts.map((o,j)=>`<button class="quiz-option" onclick="selectAnswer(${i},${j},this)">${o}</button>`).join('')}
    </div>`;
    container.appendChild(div);
  });
  checkAllAnswered();
}

function selectAnswer(qi, ai, btn) {
  const opts = document.querySelectorAll(`#opts-${qi} .quiz-option`);
  opts.forEach(b => b.classList.remove('selected'));
  btn.classList.add('selected');
  btn.style.borderColor = 'var(--purple)';
  btn.style.background = 'var(--purple-light)';
  state.quizAnswers[qi] = ai;
  opts.forEach(b => { if(!b.classList.contains('selected')) { b.style.borderColor=''; b.style.background=''; } });
  checkAllAnswered();
}

function checkAllAnswered() {
  const allDone = state.quizQuestions.length > 0 && Object.keys(state.quizAnswers).length === state.quizQuestions.length;
  document.getElementById('quiz-submit-btn').disabled = !allDone;
}

function submitQuiz() {
  let score = 0;
  state.quizQuestions.forEach((q, i) => {
    const opts = document.querySelectorAll(`#opts-${i} .quiz-option`);
    opts.forEach((b, j) => {
      b.disabled = true;
      if(j === q.ans) b.classList.add('correct');
      else if(j === state.quizAnswers[i] && j !== q.ans) b.classList.add('wrong');
    });
    if(state.quizAnswers[i] === q.ans) score++;
  });
  state.quizScore = score * 10;
  document.getElementById('quiz-submit-btn').textContent = `${score}/5 정답! → 놀이문제 풀기`;
  document.getElementById('quiz-submit-btn').disabled = false;
  document.getElementById('quiz-submit-btn').onclick = () => goToStep(4);
  updateTopBars();
}

async function initPlay() {
  updateTopBars();
  state.selectedLeft = null; state.selectedRight = null;
  state.matchedPairs = new Set(); state.matchWrong = 0;
  document.getElementById('play-loading').style.display = 'block';
  document.getElementById('play-content').style.display = 'none';
  document.getElementById('play-next-btn').disabled = true;
  
  const label = state.lang==='en' ? '한국어 단어와 영어 뜻을 짝지어 보세요! 🎯' : 'Сопоставьте корейские слова с русскими значениями! 🎯';
  document.getElementById('play-instruction').textContent = label;

  let pairs = [];
  try {
    const res = await callClaude(`
다음 동화에서 5개의 단어를 골라 짝짓기 게임용 JSON을 만들어주세요.
동화: "${STORY.text}"
대상 모국어: ${state.lang==='en'?'영어':'러시아어'}

다음 형식으로만 응답해주세요:
[
  {"ko":"한국어단어","trans":"${state.lang==='en'?'영어번역':'русский перевод'}"},
  ...
]
5개만 만들어주세요.
`);
    const clean = res.replace(/\`\`\`json|\`\`\`/g,'').trim();
    const parsed = JSON.parse(clean);
    if(Array.isArray(parsed)&&parsed.length>=5) pairs = parsed.slice(0,5);
    else throw new Error('invalid');
  } catch(e) {
    pairs = MATCH_FALLBACK.map(p => ({ko:p.ko, trans:state.lang==='en'?p.en:p.ru}));
  }
  state.matchPairs = pairs;
  renderMatchGame(pairs);
}

function renderMatchGame(pairs) {
  document.getElementById('play-loading').style.display = 'none';
  document.getElementById('play-content').style.display = 'block';
  
  const leftCol = document.getElementById('left-col');
  const rightCol = document.getElementById('right-col');
  leftCol.innerHTML = ''; rightCol.innerHTML = '';
  
  const shuffledRight = [...pairs].sort(()=>Math.random()-0.5);
  
  pairs.forEach((p, i) => {
    const lb = document.createElement('div');
    lb.className = 'match-item'; lb.textContent = p.ko;
    lb.dataset.idx = i; lb.dataset.side = 'left';
    lb.onclick = () => selectMatchItem(lb, 'left', i);
    leftCol.appendChild(lb);
  });
  
  shuffledRight.forEach((p, i) => {
    const rb = document.createElement('div');
    rb.className = 'match-item'; rb.textContent = p.trans;
    rb.dataset.idx = pairs.indexOf(p); rb.dataset.side = 'right';
    rb.onclick = () => selectMatchItem(rb, 'right', pairs.indexOf(p));
    rightCol.appendChild(rb);
  });
}

function selectMatchItem(el, side, idx) {
  if(el.classList.contains('matched-correct')) return;
  
  if(side==='left') {
    document.querySelectorAll('#left-col .match-item').forEach(b => {
      if(!b.classList.contains('matched-correct')) b.classList.remove('selected');
    });
    el.classList.add('selected');
    state.selectedLeft = {el, idx};
  } else {
    document.querySelectorAll('#right-col .match-item').forEach(b => {
      if(!b.classList.contains('matched-correct')) b.classList.remove('selected');
    });
    el.classList.add('selected');
    state.selectedRight = {el, idx};
  }
  
  if(state.selectedLeft && state.selectedRight) {
    const l = state.selectedLeft, r = state.selectedRight;
    if(l.idx === r.idx) {
      l.el.classList.remove('selected'); r.el.classList.remove('selected');
      l.el.classList.add('matched-correct'); r.el.classList.add('matched-correct');
      state.matchedPairs.add(l.idx);
      state.selectedLeft = null; state.selectedRight = null;
      if(state.matchedPairs.size === state.matchPairs.length) {
        const sc = Math.max(0, 5 - state.matchWrong) * 10;
        state.matchScore = sc;
        document.getElementById('match-score-display').style.display = 'block';
        document.getElementById('match-score-txt').textContent = `🎯 ${state.matchedPairs.size}/5 완성! ${sc}점`;
        document.getElementById('play-next-btn').disabled = false;
        updateTopBars();
      }
    } else {
      state.matchWrong++;
      l.el.classList.add('matched-wrong'); r.el.classList.add('matched-wrong');
      setTimeout(() => {
        l.el.classList.remove('matched-wrong','selected');
        r.el.classList.remove('matched-wrong','selected');
        state.selectedLeft = null; state.selectedRight = null;
      }, 800);
    }
  }
}

async function callClaude(prompt) {
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 1000,
      messages: [{role:'user', content: prompt}]
    })
  });
  const data = await res.json();
  return data.content.map(c=>c.text||'').join('');
}
